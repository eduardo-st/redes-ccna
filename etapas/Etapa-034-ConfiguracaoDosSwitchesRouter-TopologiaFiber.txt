!PRIMEIRA ETAPA: Configuração Base Switch Layer 2 2960
enable
clock set 14:00:00 26 May 2020
!Configuração do Histórico de comandos e Monitoramento dos Log no Terminal
terminal history size 256
terminal monitor
	configure terminal
		hostname sw-l2-2960-5
		service password-encryption
		service timestamps log datetime msec
		logging on
		logging host 192.168.3.1
		logging trap debugging
		snmp-server community ptiro ro
		snmp-server community ptirw rw
		cdp run
		lldp run
		ip domain-lookup
		ip domain-name vaamonde.pti
		ip name-server 192.168.3.1
		!OBSERVAÇÃO: as SVI dos Switches serão configuradas utilizando a VLAN-99
		ip default-gateway 192.168.99.254
		clock timezone BR -3
		ntp authenticate
		ntp authentication-key 123456 md5 vaamonde
		ntp server 192.168.3.1
		banner motd #AVISO: acesso autorizado somente a funcionarios#
		enable secret vaamonde@pti
		username robson secret vaamonde@pti
		username vaamonde password vaamonde@pti
		username admin privilege 15 secret vaamonde@pti
		!ADENDO: pode copiar o script completo e aplicar nos equipamentos, usando a linha
		!de geração das chaves do módulo do SSH Server completa
		crypto key generate rsa usage-keys modulus 1024
		ip ssh version 2
		ip ssh time-out 60
		ip ssh authentication-retries 2
		cdp run
		lldp run
		line console 0
			login local
			password vaamonde@pti
			logging synchronous
			exec-timeout 5 30
			exit	
		line vty 0 4
			login local
			password vaamonde@pti
			logging synchronous
			exec-timeout 5 30
			transport input ssh
			end
write

!QUARTA ETAPA: Configuração Base Router 2911
enable
clock set 14:00:00 26 May 2020
terminal history size 256
terminal monitor
	configure terminal
		hostname rt-2911-1
		service password-encryption
		service timestamps log datetime msec
		service timestamps debug datetime msec
		logging on
		logging host 192.168.3.1
		logging trap debugging
		login on-success log
		login on-failure log
		logging userinfo
		snmp-server community ptiro ro
		snmp-server community ptirw rw
		cdp run
		lldp run
		ip domain-lookup
		ip domain-name vaamonde.pti
		ip name-server 192.168.3.1
		ntp authenticate
		ntp authentication-key 123456 md5 vaamonde
		ntp server 192.168.3.1
		ntp update-calendar
		banner motd #AVISO: acesso autorizado somente a funcionarios#
		security passwords min-length 10
		enable secret vaamonde@pti
		username robson secret vaamonde@pti
		username vaamonde password vaamonde@pti
		username admin privilege 15 secret vaamonde@pti
		crypto key generate rsa usage-keys modulus 1024
		ip ssh version 2
		ip ssh time-out 60
		ip ssh authentication-retries 2
		login block-for 120 attempts 2 within 60
		cdp run
		lldp run
		line console 0
			login local
			password vaamonde@pti
			logging synchronous
			exec-timeout 5 30
			exit
		line aux 0
			login local
			password vaamonde@pti
			logging synchronous
			exec-timeout 5 30
			speed 115200
			flowcontrol hardware
			exit
		line vty 0 4
			login local
			password vaamonde@pti
			logging synchronous
			exec-timeout 5 30
			transport input ssh
			end
write
